	\paragraph{Aufgabe 11}\mbox{}\\
	Wie oft wird die folgende Schleife ausgeführt und warum?
	\begin{lstlisting}
	int i = 10;
	do { 
		i = i - 3;
	} while (i  > 5);
	\end{lstlisting}
	
	\paragraph{Aufgabe 12}\mbox{}\\
	Welche Zahlen werden bei diesen Schleifen ausgegeben? Und wie oft werden diese Ausgeführt?
	\\\\
	a) \begin{lstlisting}
	for( int i=1; i <= 10; i++ ){
		System.out.println( i );
	} 
	\end{lstlisting}
	b) \begin{lstlisting}
	for( int i=1; i <= 10; i = i+2 ){
		System.out.println( i );
	}  
	\end{lstlisting}
	c) \begin{lstlisting}
	for( int i=1; i <= 10; i = i*2 ){
		System.out.println( i );
	} 
	\end{lstlisting}
	d) \begin{lstlisting}
	for( int i=1; i < 10; i = i+2 ){
		if (i >= 5) {
			System.out.println( i );
			i--;
		} else {
			System.out.println( i );
		} 
	} 
	\end{lstlisting}
	\newpage
	e) \begin{lstlisting}
	int i = 0;
	do { 
		if (i < 4) {
			System.out.println( i );
		}
		if (i > 4) {
			System.out.println( i );
		} else {
			i--;
		} 
		i = i+2;
	} while (i  < 10);  
	\end{lstlisting}
